<script>
function change(t) {
	//랜덤, 불규칙한 수, 예측불가능한 수, 난수
	//Math.random() 0.0 ~ 1.0사이의 값을 돌려준다.
	r = parseInt(Math.random()*256);
	g = parseInt(Math.random()*256);
	b = parseInt(Math.random()*256);
	
	//t.style.color = 'rgb(' + a + ',' + b + ',' + c + ')';
	//alert(r + "," + g + "," + b);
	
	//r,g,b를 16진수로 변환 - toString(16) 메소드 사용.
	r = r.toString(16);
	g = g.toString(16);
	b = b.toString(16);
	
	t.style.color = "#" + r + g + b;
	//alert(r + "," + g + "," + b);
}
</script>
